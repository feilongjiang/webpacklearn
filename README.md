# webpacklearn
快速开始一个webpack 练手项目
## 项目初始化过程 ，以后不用下面这么多的命令，直接npm install即可
~~~bash
mkdir webpack-demo && cd webpack-demo
npm init -y
npm install webpack webpack-cli --save-dev 
npm install --save-dev style-loader css-loader // 样式加载
npm install --save-dev file-loader//资源加载
npm install --save-dev csv-loader xml-loader //资源加载
npm install --save-dev html-webpack-plugin // html打包插件
npm install clean-webpack-plugin --save-dev // 清理插件
npm install --save-dev webpack-dev-server //开发服务器
npm install --save-dev webpack-merge // 配置合并
npm install --save-dev uglifyjs-webpack-plugin // 压缩
npm install --save-dev babel-loader@7 babel-core babel-preset-es2016 // 
npm install --save-dev typescript ts-loader
~~~
